## 求1-n的最小公倍数

这个东西就是靠你一个质数筛+Bigint：

```js
...
rl.on("line", (line) => {
	let n = parseInt(line);
	let isPrime = Array(n+1).fill(true);
	isPrime[0] = isPrime[1] = false;
	
	for(let i = 2 ; i*i <= n ; i++) {
		if(isPrime[i]) {
			for(let j= i*i ; j <= n; j+=i) {
				isPrime[j] = false;
			}
		}
	}
	let res = 1n;
	for(let i = 2; i<= n ; i++) {
		if(isPrime[i]) {
			let tmp = BigInt(i);
			while(tmp*BigInt(i) <= BigInt(n)) {
				tmp*= BigInt(i);
			}
			res = (res * tmp) % MOD
		}
	}
	console.log(res.toString());
	rl.close();
})

```
