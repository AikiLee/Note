
## leetcode构造链表

直接使用function来构造类：

```js
//构造链表节点
const ListNode = function (val, next) {
    this.val = val || 0;
	this.next = next || null;
}

//将数组转换为链表
const getList = function makeList(array) {
  let head = new ListNode();
  let cur = head;
  for(let i = 0; i < array.length; i++){
    let node = new ListNode(array[i]);
    cur.next = node;
    cur = cur.next;
  }
  return head.next;
}
```
